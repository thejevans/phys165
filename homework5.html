import scipy.linalg as la
from IPython.display import display, Math

a = [[2, 1, 4], [1, 4, 3], [4, 3, 1]]
b = [[3, 1, 2], [1, 4, 0], [2, 0, 2]]

eigs = []

for mat in a, b:
    values, vectors = la.eigh(mat)
    eig = []
    
    for val, vec in zip(values, vectors):
        val_str = r'{:.4g}'.format(val)
        vec_str = r'\begin{{bmatrix}} {:.4g} \\ {:.4g} \\ {:.4g} \end{{bmatrix}}'.format(*vec)
        eig.append([val_str, vec_str])
    
    eigs.append(eig)

eigs = np.array(eigs)

output = ''.join([r'\begin{array}{cccc}\hline',
                  r'\hat{A} & & \hat{B}\\',
                  r'\lambda & v & \lambda & v \\\hline',
                  *['&'.join([eigs[0,i,0], eigs[0,i,1], eigs[1,i,0], eigs[1,i,1], r' \\']) for i in range(3)],
                  r'\hline\end{array}'])

display(Math(output))

import matplotlib.pyplot as plt
from scipy.integrate import quad
import numpy as np
from matplotlib import rcParams

rcParams['figure.dpi'] = 300

def f(x, a): return x/np.sqrt(x**2 + a**2)

xs     = np.linspace(0, 10, 1000)
a_list = [1, 3, 5, 10]

for a in a_list:
    int_f = np.array([quad(f, 0, x, args = (a,)) for x in xs])
    plt.plot(xs, int_f[:, 0], label = 'a = {:}'.format(a))

plt.xlabel('x')
plt.ylabel(r'$\int\frac{x dx}{\sqrt{x^2+a^2}}$')
plt.legend()
plt.show()
